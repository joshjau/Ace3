--[[
====================================================================
CombatLogGetCurrentEventInfo
Returns current combat log event data with variable return values
Introduced: Patch 8.0 (Battle for Azeroth)
Last Changed: Patch 9.0 (Shadowlands) - Added environmental type tracking
====================================================================
]]

--[[
Version History:
- 8.0.1 (2018-07-17): Initial implementation
- 8.1.0 (2018-12-11): Added isOffHand parameter for melee events
- 9.0.1 (2020-10-13): Added environmental damage type differentiation
- 10.0.0 (2022-10-25): Dragonflight - No structural changes
- 11.0.0 (2024-08-26): The War Within - Current implementation
]]

--[[
Base Parameters (indices 1-11):
1.  timestamp         (number)  - Time of event (GetTime() format)
2.  eventType         (string)  - Full event type (e.g., "SPELL_DAMAGE")
3.  hideCaster        (boolean) - Whether source is hidden
4.  sourceGUID        (string)  - Full GUID of source entity
5.  sourceName        (string)  - Name of source unit
6.  sourceFlags       (number)  - Unit flags for source (bitmask)
7.  sourceRaidFlags   (number)  - Raid target index for source
8.  destGUID          (string)  - Full GUID of destination entity
9.  destName          (string)  - Name of destination unit
10. destFlags         (number)  - Unit flags for destination (bitmask)
11. destRaidFlags     (number)  - Raid target index for destination

Extra Parameters (indices 12+ - event-specific):
12. spellId           (number)  - Spell ID for spell-related events
13. spellName         (string)  - Localized spell name
14. spellSchool       (number)  - Spell school bitmask
15. amount            (number)  - Damage/healing amount
16. overkill          (number)  - Excess damage beyond target's health
17. school            (number)  - Damage school bitmask
18. resisted          (number)  - Amount resisted
19. blocked           (number)  - Amount blocked
20. absorbed          (number)  - Amount absorbed
21. critical          (boolean) - Whether hit was critical
22. glancing          (boolean) - Whether hit was glancing
23. crushing          (boolean) - Whether hit was crushing (Legacy mechanic, removed in BFA)
24. isOffHand         (boolean) - For melee/swing events
]]

--[[
Common Event Subtypes and Parameter Mapping:
• SPELL_DAMAGE:       12-22 (spellId, ..., critical)
• SWING_DAMAGE:       12-20 (amount, ..., critical)
• SPELL_HEAL:         12-16 (spellId, ..., absorbed, critical)
• SPELL_AURA_APPLIED: 12-14 (spellId, auraType, amount)
• ENVIRONMENTAL_DAMAGE: 12-18 (environmentalType, ..., critical)
]]

--[[
Usage Example:
local function OnCombatEvent()
    local timestamp, eventType, _, 
          sourceGUID, sourceName, _, _, 
          destGUID, destName = CombatLogGetCurrentEventInfo()
    
    if eventType == "SPELL_DAMAGE" then
        local spellId, _, _, amount, _, _, critical = select(12, CombatLogGetCurrentEventInfo())
        -- Process damage event
    end
end
]]

--[[
Best Practices:
1. Always check eventType before accessing extra parameters
2. Use UnitGUID("player") for player comparison instead of name
3. Prefer spell IDs over names for spell identification
4. Use bit.band() with sourceFlags/destFlags for unit type checks
]]

--[[
Error Handling:
- Returns nil values for invalid parameter indices
- Extra parameters vary by event type (verify with subevent)
- Unit names may be nil for hidden/unknown entities
]]

--[[
References:
• https://warcraft.wiki.gg/wiki/API_CombatLogGetCurrentEventInfo
• In-game FrameXML/CombatLog.lua

--[[
====================================================================
GetTimePreciseSec
Returns high-precision monotonic timestamp in seconds
Introduced: Patch 7.3.0 (Legion)
Last Changed: Patch 9.0 (Shadowlands) - Behavior stabilized
====================================================================
]]

--[[
Version History:
- 7.3.0 (2017-08-29): Initial implementation
- 8.0.1 (2018-07-17): Fixed macOS/Linux timing source
- 9.0.1 (2020-10-13): Guaranteed monotonic behavior
- 11.0.0 (2024-08-26): The War Within - Current implementation
]]

--[[
Parameters: None

Returns:
seconds (number) - Monotonic timestamp in seconds with µs precision
]]

--[[
Key Features:
1. Monotonic (never decreases)
2. Microsecond resolution (actual precision varies by hardware)
3. Not affected by system clock changes
4. First call initializes counter (returns 0.0 until first frame render)
]]

--[[
Best Practices:
1. Use for combat logging/ability timing
2. Initialize early to avoid initial 0.0 value
3. Compare deltas rather than absolute values
4. Combine with C_Timer.NewTicker for precision loops
5. Avoid excessive calls (>1000/sec may impact performance)
]]

--[[
Example:
local start = GetTimePreciseSec()
PerformCombatAction()
local duration = GetTimePreciseSec() - start
print(string.format("Action took %.3f ms", duration * 1000))
]]

--[[
Parameters:

Returns:
spellInfo (table) - Structured spell data or nil if not found

  .iconID         (number)  - Current display icon (fileID)
  .originalIconID (number)  - Base spell icon (unaffected by overrides)
  .castTime       (number)  - Cast duration in milliseconds (0=instant)
  .minRange       (number)  - Minimum range requirement (0=no minimum)
  .maxRange       (number)  - Maximum effective range (0=unlimited)
  .spellID        (number)  - Spell ID of queried spell
]]

--[[
Best Practices:
1. Prefer spell IDs over names for reliability
2. Check nil returns before accessing fields
3. Use SpellMixin:ContinueOnSpellLoad() for async data loading
4. Cache results for frequently accessed spells
5. Use originalIconID for talent-modified spell tracking
]]

--[[
Example:
local spellData = C_Spell.GetSpellInfo(2061) -- Flash Heal
if spellData then
    print(string.format("%s (ID %d) Range: %d yards", 
          spellData.name, spellData.spellID, spellData.maxRange))
end
]]

- "partyN"          : Party members (1-4)

- "arenaN"          : Arena opponents (1-5)
- "bossN"           : Encounter bosses (1-8)

- "softenemy"       : Auto-targeted hostile unit
- "softfriend"      : Auto-targeted friendly unit
- "softinteract"    : Auto-targeted interactable object

- "anyfriend"       : Current target OR softfriend

--[[
Target Chaining:
Append "target" to any UnitId for target hierarchy:
- "targettarget"    : Target's target
- "pettarget"       : Pet's target
- "party1target"    : First party member's target

Version History:

====================================================================
UnitName/UnitFullName/UnitNameUnmodified
Returns name and realm information for units
Introduced:
- UnitName: Patch 1.0.0 (Vanilla)
- UnitFullName: Patch 5.4.2 (Mists of Pandaria)
- UnitNameUnmodified: Patch 9.0.2 (Shadowlands)
====================================================================
]]

--[[
Parameters:
unit (string) - Valid UnitId ("player", "target", etc.)

Returns:
name (string?)  - Unit name or nil if invalid
realm (string?) - Normalized realm name or nil if same realm
]]

--[[
Version History:
- 1.0.0 (2004-11-23): UnitName initial implementation
- 1.12.0 (2006-08-22): Added realm return value
- 5.4.2 (2013-09-10): UnitFullName introduced
- 9.0.2 (2020-11-17): UnitNameUnmodified added
- 11.0.5 (2024-09-10): Current TWW implementation
]]

--[[
Function Differences:
1. UnitName()
   - Returns realm only for cross-realm units
   - Affected by name-modifying effects

2. UnitFullName()
   - Always returns realm for "player" unit
   - Consistent cross-realm handling

3. UnitNameUnmodified()
   - Returns base name ignoring temporary modifications
   - Unaffected by effects like [Lifegiving Seed]
]]

--[[
Best Practices:
1. Use UnitFullName() for player name/realm consistency
2. Prefer UnitNameUnmodified() for spell/target tracking
3. Always check nil returns before using names
4. Use Ambiguate() for cross-realm name formatting
5. Combine with UnitGUID() for persistent identification
]]

--[[
Examples:
-- Basic name check
local name, realm = UnitName("target")
if name then
    print("Target: "..(realm and name.."-"..realm or name))
end

-- Player name with realm
local playerName, playerRealm = UnitFullName("player")
print("Player: "..playerName.."@"..playerRealm)
]]

--[[
GetUnitName Formatting:
formattedName = GetUnitName(unit, showServerName)
- showServerName (boolean):
  true: "Name-Realm" for cross-realm
  false: "Name (*)" for coalesced realms

Example:
print(GetUnitName("target", true))  -- "Valnaar-EarthenRing"
print(GetUnitName("target", false)) -- "Valnaar (*)"
]]

--[[
Related Functions:
- Ambiguate()          : Prevents name collisions
- UnitGUID()           : Unique persistent identifier
- GetPlayerInfoByGUID(): Cross-realm player data
]]
